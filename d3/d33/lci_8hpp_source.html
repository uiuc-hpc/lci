744/***********************************************************************
745 * Overloading graph_add_node for functor
746 **********************************************************************/
747#if __cplusplus >= 201703L
748template <typename T>
749status_t graph_execute_op_fn(void* value)
750{
751 auto op = static_cast<T*>(value);
752 using result_t = std::invoke_result_t<T>;
753
754 if constexpr (std::is_same_v<result_t, status_t>) {
755 status_t result = (*op)();
756 return result;
757 } else if constexpr (std::is_same_v<result_t, errorcode_t>) {
758 errorcode_t result = (*op)();
759 return result;
760 } else {
761 (*op)();
762 return errorcode_t::done;
763 }
764}
765#else
766// Specialization for status_t return type
767template <typename T>
768typename std::enable_if<
769 std::is_same<typename std::result_of<T()>::type, status_t>::value,
770 status_t>::type
771graph_execute_op_fn(void* value)
772{
773 auto op = static_cast<T*>(value);
774 status_t result = (*op)();
775 return result;
776}
777
778// Specialization for errorcode_t return type
779template <typename T>
780typename std::enable_if<
781 std::is_same<typename std::result_of<T()>::type, errorcode_t>::value,
782 status_t>::type
783graph_execute_op_fn(void* value)
784{
785 auto op = static_cast<T*>(value);
786 errorcode_t result = (*op)();
787 return result;
788}
789
790// Specialization for all other return types
791template <typename T>
792typename std::enable_if<
793 !std::is_same<typename std::result_of<T()>::type, status_t>::value &&
794 !std::is_same<typename std::result_of<T()>::type, errorcode_t>::value,
795 status_t>::type
796graph_execute_op_fn(void* value)
797{
798 auto op = static_cast<T*>(value);
799 (*op)();
800 return errorcode_t::done;
801}
802#endif
803
804template <typename T>
805void graph_free_op_fn(void* value)
806{
807 auto op = static_cast<T*>(value);
808 delete op;
809}

819template <typename T>
820graph_node_t graph_add_node_op(comp_t graph, const T& op)
821{
822 graph_node_run_cb_t wrapper = graph_execute_op_fn<T>;
823 T* fn = new T(op);
824 graph_node_free_cb_t free_cb = graph_free_op_fn<T>;
825 auto ret = graph_add_node_x(graph, wrapper)
826 .value(reinterpret_cast<void*>(fn))
827 .free_cb(free_cb)();
828 fn->user_context(ret);
829 return ret;
830}

887// copy constructor
888inline data_t::data_t(const data_t& other)
889{
890 set_type(other.get_type());
891 bool own_data = other.get_own_data();
892 set_own_data(own_data);
893 if (own_data)
894 fprintf(stderr, "Copying buffer with own_data=true is not recommended\n");
895 if (other.is_scalar()) {
896 scalar.size = other.scalar.size;
897 memcpy(scalar.data, other.scalar.data, scalar.size);
898 } else if (other.is_buffer()) {
899 buffer = other.buffer;
900 if (own_data) {
901 buffer.base = malloc(other.buffer.size);
902 memcpy(buffer.base, other.buffer.base, other.buffer.size);
903 }
904 } else if (other.is_buffers()) {
905 buffers.count = other.buffers.count;
906 buffers.buffers = new buffer_t[buffers.count];
907 for (size_t i = 0; i < buffers.count; i++) {
908 buffers.buffers[i] = other.buffers.buffers[i];
909 if (own_data) {
910 buffers.buffers[i].base = malloc(other.buffers.buffers[i].size);
911 memcpy(buffers.buffers[i].base, other.buffers.buffers[i].base,
912 other.buffers.buffers[i].size);
913 }
914 }
915 }

997template <typename T>
998T data_t::get_scalar(/* always copy semantic */) const
999{
1000 // We still keep the ownership of the data
1001 if (is_buffer()) {
1002 if (buffer.size != sizeof(T)) {
1003 throw std::runtime_error("Buffer size does not match scalar size");
1004 }
1005 return *reinterpret_cast<const T*>(buffer.base);
1006 } else if (is_scalar()) {
1007 if (sizeof(T) > scalar.size) {
1008 throw std::runtime_error("No enough data to fit the scalar.");
1009 }
1010 return *reinterpret_cast<const T*>(scalar.data);
1011 } else {
1012 throw std::runtime_error("Cannot convert to a scalar");
1013 }
1014}

1016inline buffer_t data_t::get_buffer(get_semantic_t semantic)
1017{
1018 buffer_t ret;
1019 if (is_scalar()) {
1020 if (semantic == get_semantic_t::move || semantic == get_semantic_t::copy) {
1021 ret.size = scalar.size;
1022 ret.base = malloc(scalar.size);
1023 memcpy(ret.base, scalar.data, scalar.size);
1024 } else {
1025 ret = buffer_t(scalar.data, scalar.size);
1026 }
1027 } else if (is_buffer()) {
1028 if (semantic == get_semantic_t::copy && get_own_data()) {
1029 ret.size = buffer.size;
1030 ret.base = malloc(buffer.size);
1031 memcpy(ret.base, buffer.base, buffer.size);
1032 } else {
1033 ret = buffer_t(buffer.base, buffer.size, buffer.mr);
1034 }
1035 } else {
1036 throw std::runtime_error("Cannot convert to a buffer");
1037 }
1038 if (semantic == get_semantic_t::move) {
1039 set_own_data(false);
1040 set_type(LCI_DATA_TYPE_NONE);
1041 }
1042 return ret;
1043}

1045inline size_t data_t::get_buffers_count() const

1048 throw std::runtime_error("Not a buffers");

1052
1053inline buffers_t data_t::get_buffers(get_semantic_t semantic)
1054{
1055 if (!is_buffers()) {
1056 throw std::runtime_error("Not buffers");
1057 }
1058 if (semantic == get_semantic_t::move) {
1059 set_own_data(false);
1060 set_type(LCI_DATA_TYPE_NONE);
1061 }
1062 buffers_t ret;
1063 for (size_t i = 0; i < buffers.count; i++) {
1064 if (semantic == get_semantic_t::copy && get_own_data()) {
1065 buffer_t buffer;
1066 buffer.size = buffers.buffers[i].size;
1067 buffer.base = malloc(buffer.size);
1068 memcpy(buffer.base, buffers.buffers[i].base, buffer.size);
1069 ret.push_back(buffer);
1070 } else {
1071 ret.push_back(buffers.buffers[i]);
1072 }
1073 }
1074 return ret;
1075}
