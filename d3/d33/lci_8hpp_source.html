628#if __cplusplus >= 201703L
629template <typename T>
630status_t graph_execute_op_fn(void* value)
631{
632 auto op = static_cast<T*>(value);
633 using result_t = std::invoke_result_t<T>;
634
635 if constexpr (std::is_same_v<result_t, status_t>) {
636 status_t result = (*op)();
637 return result;
638 } else if constexpr (std::is_same_v<result_t, errorcode_t>) {
639 errorcode_t result = (*op)();
640 return result;
641 } else {
642 (*op)();
643 return errorcode_t::done;
644 }
645}
646#else
647// Specialization for status_t return type
648template <typename T>
649typename std::enable_if<
650 std::is_same<typename std::result_of<T()>::type, status_t>::value,
651 status_t>::type
652graph_execute_op_fn(void* value)
653{
654 auto op = static_cast<T*>(value);
655 status_t result = (*op)();
656 return result;
657}
658
659// Specialization for errorcode_t return type
660template <typename T>
661typename std::enable_if<
662 std::is_same<typename std::result_of<T()>::type, errorcode_t>::value,
663 status_t>::type
664graph_execute_op_fn(void* value)
665{
666 auto op = static_cast<T*>(value);
667 errorcode_t result = (*op)();
668 return result;
669}
670
671// Specialization for all other return types
672template <typename T>
673typename std::enable_if<
674 !std::is_same<typename std::result_of<T()>::type, status_t>::value &&
675 !std::is_same<typename std::result_of<T()>::type, errorcode_t>::value,
676 status_t>::type
677graph_execute_op_fn(void* value)
678{
679 auto op = static_cast<T*>(value);
680 (*op)();
681 return errorcode_t::done;
682}
683#endif
684
685template <typename T>
686void graph_free_op_fn(void* value)
687{
688 auto op = static_cast<T*>(value);
689 delete op;
690}

700template <typename T>
701graph_node_t graph_add_node_op(comp_t graph, const T& op)
702{
703 graph_node_run_cb_t wrapper = graph_execute_op_fn<T>;
704 T* fn = new T(op);
705 graph_node_free_cb_t free_cb = graph_free_op_fn<T>;
706 auto ret = graph_add_node_x(graph, wrapper)
707 .value(reinterpret_cast<void*>(fn))
708 .free_cb(free_cb)();
709 fn->user_context(ret);
710 return ret;
711}
