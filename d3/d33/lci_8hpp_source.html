594 * Overloading graph_add_node for functor

596#if __cplusplus >= 201703L
597template <typename T>
598status_t graph_execute_op_fn(void* value)
599{
600 auto op = static_cast<T*>(value);
601 using result_t = std::invoke_result_t<T>;
602
603 if constexpr (std::is_same_v<result_t, status_t>) {
604 status_t result = (*op)();
605 return result;
606 } else if constexpr (std::is_same_v<result_t, errorcode_t>) {
607 errorcode_t result = (*op)();
608 return result;
609 } else {
610 (*op)();
611 return errorcode_t::done;
612 }
613}
614#else
615// Specialization for status_t return type
616template <typename T>
617typename std::enable_if<
618 std::is_same<typename std::result_of<T()>::type, status_t>::value,
619 status_t>::type
620graph_execute_op_fn(void* value)
621{
622 auto op = static_cast<T*>(value);
623 status_t result = (*op)();
624 return result;
625}
626
627// Specialization for errorcode_t return type
628template <typename T>
629typename std::enable_if<
630 std::is_same<typename std::result_of<T()>::type, errorcode_t>::value,
631 status_t>::type
632graph_execute_op_fn(void* value)
633{
634 auto op = static_cast<T*>(value);
635 errorcode_t result = (*op)();
636 return result;
637}
638
639// Specialization for all other return types
640template <typename T>
641typename std::enable_if<
642 !std::is_same<typename std::result_of<T()>::type, status_t>::value &&
643 !std::is_same<typename std::result_of<T()>::type, errorcode_t>::value,
644 status_t>::type
645graph_execute_op_fn(void* value)
646{
647 auto op = static_cast<T*>(value);
648 (*op)();
649 return errorcode_t::done;
650}
651#endif
652
653template <typename T>
654void graph_free_op_fn(void* value)
655{
656 auto op = static_cast<T*>(value);
657 delete op;
658}
659
668template <typename T>
669graph_node_t graph_add_node_op(comp_t graph, const T& op)
670{
671 graph_node_run_cb_t wrapper = graph_execute_op_fn<T>;
672 T* fn = new T(op);
673 graph_node_free_cb_t free_cb = graph_free_op_fn<T>;
674 auto ret = graph_add_node_x(graph, wrapper)
675 .value(reinterpret_cast<void*>(fn))
676 .free_cb(free_cb)();
677 fn->user_context(ret);
678 return ret;
679}
