602/***********************************************************************
603 * Overloading graph_add_node for functor
604 **********************************************************************/
605template <typename T>
606status_t graph_execute_op_fn(void* value)
607{
608 auto op = static_cast<T*>(value);
609 using result_t = std::invoke_result_t<T>;
610
611 if constexpr (std::is_same_v<result_t, status_t>) {
612 status_t result = (*op)();
613 delete op;
614 return result;
615 } else if constexpr (std::is_same_v<result_t, errorcode_t>) {
616 errorcode_t result = (*op)();
617 delete op;
618 return result;
619 } else {
620 (*op)();
621 delete op;
622 return errorcode_t::done;
623 }
624}
625
634template <typename T>
635graph_node_t graph_add_node_op(comp_t graph, const T& op)
636{
637 graph_node_fn_t wrapper = graph_execute_op_fn<T>;
638 T* fn = new T(op);
639 auto ret =
640 graph_add_node_x(graph, wrapper).value(reinterpret_cast<void*>(fn))();
641 fn->user_context(ret);
642 return ret;
643}

672inline data_t::data_t(size_t size, allocator_base_t* allocator)
673{
674 set_type(type_t::buffer);
675 set_own_data(true);
676 if (allocator) {
677 buffer.base = allocator->allocate(size);
678 } else {
679 buffer.base = malloc(size);
680 }
681 buffer.size = size;

684inline data_t::data_t(size_t sizes[], int count, allocator_base_t* allocator)
685{
686 set_type(type_t::buffers);
687 set_own_data(true);
688 buffers.count = count;
689 buffers.buffers = new buffer_t[buffers.count];
690 for (int i = 0; i < count; i++) {
691 if (allocator) {
692 buffers.buffers[i].base = allocator->allocate(sizes[i]);
693 } else {
694 buffers.buffers[i].base = malloc(sizes[i]);
695 }
696 buffers.buffers[i].size = sizes[i];
697 buffers.buffers[i].mr = mr_t();
698 }
699}
700// copy constructor
701inline data_t::data_t(const data_t& other)
702{
703 set_type(other.get_type());
704 bool own_data = other.get_own_data();
705 set_own_data(own_data);
706 if (own_data)
707 fprintf(stderr, "Copying buffer with own_data=true is not recommended\n");
708 if (other.is_scalar()) {
709 scalar.size = other.scalar.size;
710 memcpy(scalar.data, other.scalar.data, scalar.size);
711 } else if (other.is_buffer()) {
712 buffer = other.buffer;
713 if (own_data) {
714 buffer.base = malloc(other.buffer.size);
715 memcpy(buffer.base, other.buffer.base, other.buffer.size);
716 }
717 } else if (other.is_buffers()) {
718 buffers.count = other.buffers.count;
719 buffers.buffers = new buffer_t[buffers.count];
720 for (size_t i = 0; i < buffers.count; i++) {
721 buffers.buffers[i] = other.buffers.buffers[i];
722 if (own_data) {
723 buffers.buffers[i].base = malloc(other.buffers.buffers[i].size);
724 memcpy(buffers.buffers[i].base, other.buffers.buffers[i].base,
725 other.buffers.buffers[i].size);
726 }
727 }
728 }
729}
730// move constructor
731inline data_t::data_t(data_t&& other)
732{
733 set_type(other.get_type());
734 set_own_data(other.get_own_data());
735 if (other.is_scalar()) {
736 scalar.size = other.scalar.size;
737 memcpy(scalar.data, other.scalar.data, scalar.size);
738 } else if (other.is_buffer()) {
739 buffer = other.buffer;
740 other.set_own_data(false);
741 } else if (other.is_buffers()) {
742 buffers.count = other.buffers.count;
743 buffers.buffers = other.buffers.buffers;
744 other.set_own_data(false);
745 other.buffers.buffers = nullptr;
746 }
747}

810template <typename T>
811T data_t::get_scalar(/* always copy semantic */) const
812{
813 // We still keep the ownership of the data
814 if (is_buffer()) {
815 if (buffer.size != sizeof(T)) {
816 throw std::runtime_error("Buffer size does not match scalar size");
817 }
818 return *reinterpret_cast<const T*>(buffer.base);
819 } else if (is_scalar()) {
820 if (sizeof(T) > scalar.size) {
821 throw std::runtime_error("No enough data to fit the scalar.");
822 }
823 return *reinterpret_cast<const T*>(scalar.data);
824 } else {
825 throw std::runtime_error("Cannot convert to a scalar");
826 }
827}
828
829inline buffer_t data_t::get_buffer(get_semantic_t semantic)
830{
831 buffer_t ret;
832 if (is_scalar()) {
833 if (semantic == get_semantic_t::move || semantic == get_semantic_t::copy) {
834 ret.size = scalar.size;
835 ret.base = malloc(scalar.size);
836 memcpy(ret.base, scalar.data, scalar.size);
837 } else {
838 ret = buffer_t(scalar.data, scalar.size);
839 }
840 } else if (is_buffer()) {
841 if (semantic == get_semantic_t::copy && get_own_data()) {
842 ret.size = buffer.size;
843 ret.base = malloc(buffer.size);
844 memcpy(ret.base, buffer.base, buffer.size);
845 } else {
846 ret = buffer_t(buffer.base, buffer.size, buffer.mr);
847 }
848 } else {
849 throw std::runtime_error("Cannot convert to a buffer");
850 }
851 if (semantic == get_semantic_t::move) {
852 set_own_data(false);
853 set_type(type_t::none);
854 }
855 return ret;
856}

858inline size_t data_t::get_buffers_count() const
859{
860 if (!is_buffers()) {
861 throw std::runtime_error("Not a buffers");
862 }
863 return buffers.count;
864}
865
866inline buffers_t data_t::get_buffers(get_semantic_t semantic)
867{
868 if (!is_buffers()) {
869 throw std::runtime_error("Not buffers");
870 }
871 if (semantic == get_semantic_t::move) {
872 set_own_data(false);
873 set_type(type_t::none);
874 }
875 buffers_t ret;
876 for (size_t i = 0; i < buffers.count; i++) {
877 if (semantic == get_semantic_t::copy && get_own_data()) {
878 buffer_t buffer;
879 buffer.size = buffers.buffers[i].size;
880 buffer.base = malloc(buffer.size);
881 memcpy(buffer.base, buffers.buffers[i].base, buffer.size);
882 ret.push_back(buffer);
883 } else {
884 ret.push_back(buffers.buffers[i]);
885 }
886 }
887 return ret;
888}
