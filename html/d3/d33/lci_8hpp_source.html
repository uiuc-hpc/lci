511 // copy constructor
512 inline data_t::data_t(const data_t& other)
513 {
514  type = other.type;
515  own_data = other.own_data;
516  if (own_data)
517  fprintf(stderr, "Copying buffer with own_data=true is not recommended\n");
518  if (other.is_scalar()) {
519  memcpy(scalar.data, other.scalar.data, MAX_SCALAR_SIZE);
520  } else if (other.is_buffer()) {
521  buffer = other.buffer;
522  if (own_data) {
523  buffer.base = malloc(other.buffer.size);
524  memcpy(buffer.base, other.buffer.base, other.buffer.size);
525  }
526  } else if (other.is_buffers()) {
527  buffers.count = other.buffers.count;
528  buffers.buffers = new buffer_t[buffers.count];
529  for (size_t i = 0; i < buffers.count; i++) {
530  buffers.buffers[i] = other.buffers.buffers[i];
531  if (own_data) {
532  buffers.buffers[i].base = malloc(other.buffers.buffers[i].size);
533  memcpy(buffers.buffers[i].base, other.buffers.buffers[i].base,
534  other.buffers.buffers[i].size);
535  }
536  }
537  }
538 }
539 // move constructor
540 inline data_t::data_t(data_t&& other)
541 {
542  type = other.type;
543  own_data = other.own_data;
544  if (other.is_scalar()) {
545  memcpy(scalar.data, other.scalar.data, MAX_SCALAR_SIZE);
546  } else if (other.is_buffer()) {

600 template <typename T>
601 T data_t::get_scalar(/* always copy semantic */) const
602 {
603  // We still keep the ownership of the data
604  if (is_buffer()) {
605  if (buffer.size != sizeof(T)) {
606  throw std::runtime_error("Buffer size does not match scalar size");
607  }
608  return *reinterpret_cast<const T*>(buffer.base);
609  } else if (is_scalar()) {
610  static_assert(sizeof(T) <= 23, "Scalar size is too large");
611  return *reinterpret_cast<const T*>(scalar.data);
612  } else {
613  throw std::runtime_error("Cannot convert to a scalar");
614  }
615 }

617 inline buffer_t data_t::get_buffer(get_semantic_t semantic)
618 {
619  buffer_t ret;
620  if (is_scalar()) {
621  if (semantic == get_semantic_t::move ||
622  semantic == get_semantic_t::copy) {
623  ret.size = MAX_SCALAR_SIZE;
624  ret.base = malloc(MAX_SCALAR_SIZE);
625  memcpy(ret.base, scalar.data, MAX_SCALAR_SIZE);
626  } else {
627  ret = buffer_t(scalar.data, MAX_SCALAR_SIZE);
628  }
629  } else if (is_buffer()) {
630  if (semantic == get_semantic_t::copy && own_data) {
631  ret.size = buffer.size;
632  ret.base = malloc(buffer.size);
633  memcpy(ret.base, buffer.base, buffer.size);
634  } else {
635  ret = buffer_t(buffer.base, buffer.size, buffer.mr);
636  }
637  } else {
638  throw std::runtime_error("Cannot convert to a buffer");
639  }
640  if (semantic == get_semantic_t::move) {
641  own_data = false;
642  type = type_t::none;
643  }
644  return ret;
645 }

655 inline buffers_t data_t::get_buffers(get_semantic_t semantic)
656 {
657  if (!is_buffers()) {
658  throw std::runtime_error("Not buffers");
659  }
660  if (semantic == get_semantic_t::move) {
661  own_data = false;
662  type = type_t::none;
663  }
664  buffers_t ret;
665  for (size_t i = 0; i < buffers.count; i++) {
666  if (semantic == get_semantic_t::copy && own_data) {
667  buffer_t buffer;
668  buffer.size = buffers.buffers[i].size;
669  buffer.base = malloc(buffer.size);
670  memcpy(buffer.base, buffers.buffers[i].base, buffer.size);
671  ret.push_back(buffer);
672  } else {
673  ret.push_back(buffers.buffers[i]);
674  }
675  }
676  return ret;
677 }
